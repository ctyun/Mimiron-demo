 * ```
 * @class TopBar

var TopBar = React.createClass({
    displayName: 'TopBar',
    getDefaultProps: function(){
        return{
        	/**
        	 * @property {Function} logout 注销按钮的回调函数

        	userName:"GUEST",

        }
